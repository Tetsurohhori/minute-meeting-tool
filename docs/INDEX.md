│
├── 💻 development/                     # 開発者向けガイド
│   ├── README.md
│   ├── CONTRIBUTING.md                # コントリビューションガイド
│   └── TESTING_GUIDE.md               # テストガイド
│
├── 🏛️ architecture/                   # アーキテクチャドキュメント
│   ├── README.md
│   └── PROJECT_STRUCTURE.md           # プロジェクト構造
│

    ├── README.md
    ├── DELIVERY_README.md             # 納品パッケージ概要
